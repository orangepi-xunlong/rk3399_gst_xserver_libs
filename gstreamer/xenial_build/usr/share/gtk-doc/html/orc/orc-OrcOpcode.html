typedef struct {

  OrcRuleSet rule_sets[ORC_N_RULE_SETS];

  void (*load_constant)(OrcCompiler *compiler, int reg, int size, int value);
  const char * (*get_flag_name)(int shift);
  void (*flush_cache) (OrcCode *code);

      OrcConstant *constant);

  void *_unused[5];
} OrcTarget;
